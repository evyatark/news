אלול תשפ"א, אוגוסט 2021

"ההתעמלות שצעירי ישראל עוסקים בה בארץ ישראל לחזק את גופם בשביל להיות בנים אמיצי כֹּח לאומה, היא משכללת את הכֹּח הרוחני של הצדיקים העליונים, העוסקים ביִחודים של שמות הקדושים, להרבות הבלטת האור האלֹהי בעולם, ואין גילוי אור אחד עומד בלא חבֵירו כלל" (הרֹא"ה, הרב אברהם יצחק הכהן קוק, "אורות התחִיה", פרק ל"ד). 

א.

אֱלוּל נִמְהָל בְּאוֹגוּסְט.
הַבַּיִת שָׁקֵט, אֲנִי מִתְעוֹרֵר
קוֹדֵחַ וְלֹא מֵחֹם. שׁוּב רוֹדְפוֹת אוֹתִי חַיּוֹת
הַקֹּדֶשׁ, מְעוֹרְרוֹת בַּלֵּילוֹת, מִזְדַּקְּרוֹת
בִּשְׁעוֹת הַתַּאֲוָה, פְּנֵיהֶן אוֹרוֹת בִּשְׁמֵי יְרוּשָׁלַיִם,
גּוּפָן מַדִּיף אֵימָה, בְּעֵרַת יָמִים נוֹרָאִים.
מִסִּדְקֵי הַתְּרִיסִים עָשָׁן חוֹדֵר
פּוֹשֵׁט בַּחֶדֶר, נֶאֱחָז בַּבָּשָׂר,
מְעַבֶּה אֶת עֶצֶם הַלּוּז.

קַמְתִּי בְּאַשְׁמֹרֶת
אֵינֶנִּי יוֹצֵא לִסְלִיחוֹת.
הַהִתְעַמְּלוּת שֶׁצְּעִירֵי יִשְׂרָאֵל עוֹסְקִים בָּהּ
עַל הַמָּסָךְ בְּאוֹלִימְפִּיָּאדַת טוֹקְיוֹ לֹא מְשַׁכֶּכֶת כְּאֵבִים.
הַיָּד נִשְׁלַחַת אֶל מַדַּף סִפְרֵי הַקֹּדֶשׁ
אֲנִי נֶאֱחָז בָּהֶם כְּבִתְרוּפוֹת שֶׁפָּג תָּקְפָּן.

בְּפֶתַח סֵפֶר "אוֹרוֹת"
הָרַב זָצָ"ל זוֹהֵר אֵלַי
כְּבִימֵי הַנְּעוּרִים מִי יִתְּנֵנִי
אֶקְרָא אֶל הָרְאָיָ"ה וְיַעֲנֵנִי.

  • "בלי מה": עצב הגותי בלתי מוגדר שפוך על שיריו של עמיחי חסון
  • "חָמֵש דַקות הִתְרַגַשתֶם בִּפְקֻדָה,/ חָמֵש דַקות, זוּזוּ"
  • תגיד, יש בכלל עוד משוררים היום?

ב.

בַּיְשִׁיבָה הַתִּיכוֹנִית, בִּשְׁעוֹת הַצָּהֳרַיִם הַקָּשׁוֹת
כְּשֶׁהַגּוּף הִרְפָּה עַצְמוֹ לִתְנוּמַת חָתוּל וְקוֹלוֹ הַלֵּאֶה
שֶׁל הָרַב הִפְלִיג בְּשִׁבְחֵי הַמִּצְווֹת הַתְּלוּיוֹת בְּאַרְצֵנוּ
הִתְבּוֹנַנְתִּי בִּדְיוֹקַן הָרַב זָצָ"ל, שֶׁנִּצַּב לְצַד הַלּוּחַ כְּבַקָּרַת
רוֹעֶה עֶדְרוֹ וְעֵינָיו הַטּוֹבוֹת נִחֲמוּ אֶת פָּנַי שֶׁנָּפְלוּ אַפַּיִם
אֶל הַשֻּׁלְחָן. שְׂפָתַי דּוֹבְבוּ מֵעַצְמָן: "אֶרֶץ יִשְׂרָאֵל אֵינֶנָּה
דָּבָר חִיצוֹנִי, קִנְיָן חִיצוֹנִי לָאֻמָּה".

כְּשֶׁהֵקַצְתִּי מִתַּרְדֵּמָה
הָרַב הוֹרָה לְהַצְבִּיעַ
לִמְנוֹת מַעֲלוֹת הָאֶרֶץ הַטּוֹבָה.

כְּחֹם הַיּוֹם הַכִּתָּה הִשְׁתַּתְּקָה.
יָדִי הוּנְפָה. הִתְחַלְתִּי לְדַבֵּר בְּגוּף
רִאשׁוֹן רַבִּים.

ג.

וְיָדַעְנוּ שֶׁאֶרֶץ יִשְׂרָאֵל קְדוֹשָׁה.
וְיָדַעְנוּ שֶׁאֲוִירָא דְּאֶרֶץ יִשְׂרָאֵל מַחְכִּים.
וְיָדַעְנוּ שֶׁאֶרֶץ יִשְׂרָאֵל נִקְנֵית בַּהֲלִיכָה.
וְיָדַעְנוּ שֶׁאֶרֶץ יִשְׂרָאֵל נִקְנֵית בְּיִסּוּרִים.
וְיָדַעְנוּ שֶׁאֶרֶץ יִשְׂרָאֵל הִיא קִנְיָנֵנוּ
וְאֵין לְהַפְקִירָהּ בִּידֵי זָרִים.

ד.

וְלֹא יָדַעְנוּ מַה לַּעֲשׂוֹת בַּגּוּף
וּבַשְּׂעָרוֹת שֶׁהִתְרַבּוּ בַּמִּפְשָׂעוֹת.
לַשָּׁוְא הִסְתַּרְנוּ אֶת הַזִּקְפּוֹת.
כַּפּוֹת יָדֵינוּ הִתְאַגְרְפוּ בְּתוֹךְ כִּיסִים.
קוֹלֵנוּ הִסְמִיךְ בַּחֲדָרִים סְגוּרִים.

מִדֵּי בֹּקֶר הִתְוַדַּעְנוּ לִקְרִי הַלַּיְלָה.
לֹא הֵעַזְנוּ לְסַפֵּר לְאִישׁ, לֹא הֵבַנּוּ
מָה עָלֵינוּ לְהַרְגִּישׁ. פָּקַדְנוּ בַּחֲשָׁשׁ
אֶת הָאֲתָרִים הַיְדוּעִים. נָטַלְנוּ יָדֵינוּ
שׁוּב וְשׁוּב, לְסֵרוּגִין. קָרָאנוּ מִזְמוֹרֵי תְּהִלִּים,
עֲשָׂרָה מִזְמוֹרִים יְדוּעִים, בְּחַדְרֵי הַמְתָּנָה
גִּלִּינוּ עִנְיָן מֻגְבָּר בְּעִתּוֹנֵי נָשִׁים.

עַל קִירוֹת סְנִיף בְּנֵי עֲקִיבָא צִיַּרְנוּ דְּיוֹקָן:
הָרַב זָצָ"ל זוֹהֵר בְּצִבְעֵי שֶׁמֶן בְּהִירִים.
עָלִינוּ לְשֵׁבֶט הָרֹאֶ"ה. עָמַדְנוּ בִּשְׁלָשׁוֹת
בַּמִּפְקָד לָבַשְׁנוּ מַדֵּי תּוֹרָה וַעֲבוֹדָה,
הֵנַפְנוּ דְּגָלִים. מִגַּג הַסְּנִיף רָאִינוּ:
תֵּל אָבִיב מִתְעוֹרֶרֶת, מַצִּיתָה אוֹרוֹת
מִגְדָּלֶיהָ לִקְרַאת חֲשֵׁכָה.

בְּדִמְדּוּמֵי שַׁבָּת, בְּסוֹף הַפְּעֻלָּה, שַׁרְנוּ
יַחַד בָּנוֹת וּבָנִים: "כְּשֶׁהַנְּשָׁמָה מְאִירָה
גַּם שָׁמַיִם עוֹטֵי עֲרָפֶל מְפִיקִים אוֹר נָעִים".

ה.

וְהָעוֹר הָיָה נָעִים
הַשָּׁמַיִם נְטוּלֵי עַרְפִלִּים
הַנְּשָׁמָה חִפְּשָׂה גּוּף אַחֵר לְהֵאָחֵז בּוֹ.

לְנֶגֶד עֵינֵינוּ בְּמֶרְחַק נְגִיעָה
שָׁקַע הַיּוֹם הַשְּׁבִיעִי בֵּין מִגְדְּלֵי עַזְרִיאֵלִי.
הִתְעַלַּמְנוּ. נוֹתַרְנוּ שְׁרוּיִים בְּרַעֲוָא דְּרַעֲוִין
שָׁרִים פִּיּוּט סְעֻדָּה שְׁלִישִׁית, דְּלֵית דִּינִין דִּתְקִיפִין.

יָדִי הִתְפַּתְּלָה, נִשְׁלְחָה
בְּחָסוּת הָאֲפֵלָה גִּשְּׁשָׁה
תַּחַת חֻלְצַת הַתְּנוּעָה
בְּמַגָּע זָהִיר חָשְׂפָה
הָרִים וּגְבָעוֹת
גֵּאָיוֹת וַעֲמָקִים,
אֶרֶץ מֻבְטַחַת
אֶרֶץ חֲדָשָׁה.

בְּטֶרֶם תַּרְתִּי אֶת גְּבוּלָהּ
קָרָא הַמַּדְרִיךְ:
הַבְדָּלָה!
הַבְדָּלָה!

ו.

בְּמַחֲנֵה הַקַּיִץ בְּעֵין זֵיתִים
סָפַרְנוּ צְעָדִים בֵּין מַאֲהַל
הַבָּנִים לְמַאֲהַל הַבָּנוֹת.
נוֹתַרְנוּ עֵרִים כָּל הַלַּיְלָה,
תְּשׁוּקוֹת נִפְקְחוּ בֵּין יָדֵינוּ
פָּנֵינוּ לֹא הִסְגִּירוּ.

בַּבֹּקֶר הֵגַחְנוּ מִשַּׂקֵּי הַשֵּׁנָה
לְאַט לְאַט לְאַט הֻנְּחוּ תְּפִלִּין
תַּרְנוּ אֶת אַרְצֵנוּ הַקְּדוֹשָׁה
בְּסַנְדְּלֵי שֹׁרֶשׁ מִתְפָּרְקִים.
כִּבְהִמְנוֹן הַתְּנוּעָה בְּכַפֵּינוּ חָפַנּוּ
רִגְבֵי אַדְמָתָהּ (פְּעָמִים עָלוּ בְּיָדֵינוּ
גַּם מִינֵי עֲשָׂבִים, גּוּף חָבוּי, שִׁבְרֵי
שִׁירִים, מִסְפְּרֵי טֵלֵפוֹן בֵּיתִיִּים).

אַחֲרֵי הַמַּחֲנֶה צָעַדְנוּ בְּעִקְבוֹת
הַמָּשִׁיחַ עַל שְׂפַת הַכִּנֶּרֶת בְּחוֹף דּוּגִית
סְבִיב מְדוּרָה גִּיטָרוֹת נִגְּנוּ הַדָּם גָּעַשׁ
הַגּוּף בָּעַר, חֲבֵרִים הִתְעוֹפְפוּ בִּקְפִיצַת
רֹאשׁ מִמְּצוּקֵי הַבַּזֶּלֶת שֶׁל רָמַת הַגּוֹלָן
אֶל כְּפוֹר נַחַל יָהוּדִיָּה.

כֻּלָּם קָפְצוּ לַמַּיִם. שׁוּב
פָּחַדְתִּי. יָרַדְתִּי בַּסֻּלָּם.

ז.

בִּרְחוֹב שֵׁינְקִין קָמוּ לִתְחִיָּה סְדוֹם וַעֲמוֹרָה,
גִּלָּה לָנוּ רֹאשׁ הַיְשִׁיבָה עִם פְּתִיחַת שְׁנַת הַלִּמּוּדִים.
טוֹב שֶׁמִּישֶׁהוּ יוֹדֵעַ עַל זֶה! לָחַשְׁנוּ זֶה לָזֶה
בִּשְׁעַת קִפּוּל תְּפִלִּין.

חָמַקְנוּ בְּתֹם שַׁחֲרִית.
בְּקַו שִׁשִּׁים וְאַחַת נָסַעְנוּ מִבְּנֵי בְּרַק
לְתֵל אָבִיב, יָרַדְנוּ אַחֲרֵי שְׁמוֹנֶה עֶשְׂרֵה תַּחֲנוֹת
לִרְאוֹת הַכְּצַעֲקָתָהּ.

מָצָאנוּ חֲנֻיּוֹת נַעֲלַיִם וְדוּכְנֵי מִיצִים וּפִּיצָה דַּקָּה מִקֶּמַח שְׁקֵדִים
וְעֶמְדַּת חַבָּ"ד לְהַנָּחַת תְּפִלִּין.
אֵיפֹה סְדוֹם וְהֵיכָן עֲמוֹרָה הִשְׁתּוֹמַמְנוּ
עַל סַפְסָל בְּגַן צִבּוּרִי, רְצוּף מִשְׁפָּחוֹת,
יְלָדִים וּכְלָבִים. לָמָּה לֹא יָבוֹאוּ אֵלֵינוּ הָאֲנָשִׁים וְנֵדָעֵם
נְדַבֵּר גְּלוּיוֹת, יֵשׁ אוֹ אֵין.

הָאוֹטוֹבּוּס הִתְפַּתֵּל חֲזָרָה בִּרְחוֹב הָרֹאֶ"ה עַד שַׁעַר הַיְשִׁיבָה.
הָרַב חִכָּה לָנוּ, חָקַר הֵיכָן הָיִינוּ, דָּרַשׁ לָדַעַת מָה עָשִׂינוּ בִּזְמַן
סֵדֶר בֹּקֶר בִּגְמָרָא. בָּחַן צִיצִיּוֹת בְּלִטּוּף הַגַּב (הוֹצֵאתִי
אֶת הַפְּתִילִים מִחוּץ לַחֻלְצָה כִּבְדֶרֶךְ אַגַּב). הִתְוַדֵּינוּ בְּפָנָיו:
תַּרְנוּ אַחַר לְבָבֵנוּ וְאַחַר עֵינֵינוּ. לֹא שָׂבַעְנוּ.

נִשְׁלַחְנוּ לְהַעְתִּיק מִזְמוֹר קי"ט בְּסֵפֶר תְּהִלִּים
בִּכְתַב יָדֵנוּ קי"ט פְּעָמִים. לוּלֵי תּוֹרָתְךָ עָנְשִׁי
אָז אוּלַי עָבַדְתִּי אֱלֹהִים.

ח.

לִקְרַאת הַחַגִּים יָצְאוּ שְׁלִיחִים מִירוּשָׁלַיִם
פָּרְצוּ אֶת חוֹמוֹת הַיְשִׁיבָה הַמֶּרְכָּזִית הָעוֹלָמִית
חָצוּ אֶת נְתִיבֵי אַיָּלוֹן וְיָרְדוּ אֶל שְׁפֵלַת הָאָרֶץ
לְהַרְבִּיץ בָּנוּ אַהֲבַת יִשְׂרָאֵל וְתוֹרַת יִשְׂרָאֵל
וּמְסִירוּת נֶפֶשׁ עַל אֶרֶץ יִשְׂרָאֵל בִּימֵי מֶמְשֶׁלֶת זָדוֹן.

הִתְבּוֹנַנּוּ בָּהֶם בְּסַקְרָנוּת וּבִרְתִיעָה, כְּבִקְרוֹבֵי מִשְׁפָּחָה
שֶׁבָּאוּ מֵאֶרֶץ זָרָה, שְׂפָתָם מְשֻׁבֶּשֶׁת וּמַלְבּוּשֵׁיהֶם מְשֻׁנִּים.
זִהִינוּ מִלִּים וּמִשְׁפָּטִים, תָּוֵי פָּנִים מְשֻׁתָּפִים.

עָמְדוּ הַשְּׁלִיחִים לִפְנֵי הַתֵּבָה בְּחֻלְצוֹת לְבָנוֹת בְּיוֹם חֻלִּין
וְקָרְאוּ בְּקוֹל דּוֹאֵב גַּעְגּוּעִים: "דּוֹרֵנוּ הוּא דּוֹר נִפְלָא,
דּוֹר שֶׁכֻּלּוֹ תִּמָּהוֹן!" דִּבְּרוּ בְּשֵׁם הָרַב
אַבְרָהָם יִצְחָק הַכֹּהֵן זָצָ"ל
וּבְשֵׁם בְּנוֹ הָרַב צְבִי יְהוּדָה הַכֹּהֵן זָצָ"ל
עַל תְּחִיַּת הָאֻמָּה, עַל דּוֹר גְּאֻלָּה וּבֵינְתַיִם
דּוֹר הָלַךְ וְדוֹר בָּא

צִלְצוּל צוֹרְמָנִי גָּאַל נִרְדָּמִים.
יָצָאנוּ לְהִתְאַמֵּן בְּמִגְרַשׁ הַכַּדּוּרְסַל (הִתְפַּעֲלוּ שְׁלִיחֵי מֶרְכַּז הָרַב:
"הַהִתְעַמְּלוּת שֶׁצְּעִירֵי יִשְׂרָאֵל עוֹסְקִים!"). עָלִינוּ בַּגּוֹרָל
לְשַׂחֵק מוּל יְשִׁיבַת כְּפַר הָרֹאֶ"ה בְּטוּרְנִיר הַיְשִׁיבוֹת הָאַרְצִי.
הָיִיתִי שַׂחְקַן צִיר. הֶחְזַרְתִּי כַּדּוּרִים, הֶחְמַצְתִּי
מִקַּו הָעוֹנְשִׁין. הוּבַסְנוּ כְּתָמִיד.
הַצִּיצִיּוֹת נִתְלוּ עַל עַמּוּדֵי הַסַּל סְפוּגוֹת זֵעָה.

ט.

בֵּין כֶּסֶה לְעָשׂוֹר הִתְקַיֵּם הַגּוֹמְלִין. עָלִינוּ בַּחֲשֵׁכָה
אֶל צִיּוֹן, לְהִתְעוֹרֵר בִּתְשׁוּבַת יָמִים נוֹרָאִים.

רוּחַ הַקֹּדֶשׁ נִכְנְסָה בִּבְשַׂר הַקֹּדֶשׁ
לְמַרְאֵה שְׂרִיד בֵּית מִקְדָּשֵׁנוּ. הִתְעוֹרַרְנוּ.
בְּסִיּוּרֵי סְלִיחוֹת נְעָרוֹת חָלְפוּ עַל פָּנֵינוּ
נָאוֹת וְתַמּוֹת בַּסִּמְטָאוֹת.

בִּרְחוֹב הָרַב קוּק אַבְרֵךְ אָרֹךְ מְשָׁכָנוּ אַחֲרָיו,
הוֹבִיל אוֹתָנוּ חֲדָרָיו, בְּבֵית הָרַב בִּלְחִישׁוֹת
סוֹד הֲוָיָה. מֵהַסִּפְרִיָּה פָּסַעְנוּ אֶל חֲדַר הָאוֹרְחִים.
רַק אֶל הַקֹּדֶשׁ פְּנִימָה, אֶל חֲדַר לִמּוּדוֹ
נֶאֱסַרְנוּ לָבוֹא. הִבַּטְתִּי בַּסְּפּוֹדִיק שֶׁל מָרָן
מֻנָּח יָתוֹם עַל הַשֻּׁלְחָן.

הַקְּבוּצָה הִתְקַדְּמָה
לֹא הָיָה אִישׁ עִמָּדִי
שָׁלַחְתִּי אֶת יָדִי אֶל כּוֹבַע הַפַּרְוָה
וּנְשַׁקְתִּיו.

י.

בְּבֵית הַכְּנֶסֶת הַטְּרִיפּוֹלִיטָאִי שֶׁל אָבִי
בִּרְחוֹב נֶגְבָּה פִּנַּת רְחוֹב הָרֹאֶ"ה לֹא דִּבְּרוּ
עַל הָרַב זָצָ"ל (הָאָב אוֹ הַבֵּן), "מוֹסַד הָרַב קוּק"
לֹא חִלֵּק שָׁם סִדּוּרִים. דְּיוֹקְנָאוֹת הַבַּבָּא סָאלִי
וְחַכְמֵי הַמַּגְרֶבּ עִטְּרוּ אֶת מַחֲזוֹרֵי הַכִּפּוּרִים.
טַבַּק נַעְנַע חֲרִיפָה עָבַר לַהֲרָחָה בֵּין הַמִּתְפַּלְּלִים
גְּרוֹנָם נִחָר מִפִּיּוּטִים, הֶבֶל פִּיהֶם יָבֵשׁ, מַבְאִישׁ
מִצּוֹם. עֵינָם טוֹבָה וּמִתְרַצָּה. אִישׁ לֹא בָּחַן
מַה גֹּדֶל הַכִּפָּה, מַה צֶּבַע הַחֻלְצָה.

הַיָּשָׁן הִתְחַדֵּשׁ
וְהֶחָדָשׁ הִתְקַדֵּשׁ
כְּשֶׁהַחַזָּן סִלְסֵל קַדִּישׁ
בְּמַנְגִּינַת עַבְּד אַלְוַהַאבּ.

מִחוּץ לַהֵיכָל הָרְחוֹבוֹת הָיוּ נְקִיִּים
חַמָּה עָמְדָה בְּרֹאשׁ תָּמָר נִמְהָר בֶּחָצֵר.
בְּתוֹךְ הַקָּהָל שַׁרְתִּי לְעַצְמִי:
וְנֶגְדְּךָ כָּל תַּאֲוָתִי

וּכְנֶגְדִּי עָמַד אָבִי
יָדָיו הֻנְּחוּ עַל רֹאשִׁי
תַּחַת הַטַּלִּית הַגְּדוֹלָה
בְּבִרְכַּת כֹּהֲנִים שֶׁל תְּפִלַּת נְעִילָה
נִשַּׁקְתִּי אֶת אֶצְבְּעוֹתָיו.

יא.

רָאִינוּ אוֹרוֹת הוֹלְכִים וּקְרֵבִים
בְּמוֹצָאֵי יוֹם הַכִּפּוּרִים בַּעֲשַׁן
נַרְגִּילָה בְּפִסְגַּת הַר הַבָּנִים
וּלְרֶגַע חָשַׁבְנוּ: נָגַעְנוּ בֵּאלֹהִים
אוּלַי יֵרֵד לְהָאִיר אֵלֵינוּ פָּנִים.

פָּנָסִים סִנְוְרוּ אֶת עֵינֵינוּ
חִפְּשׂוּ חֳמָרִים אֲסוּרִים.
לֹא שׁוֹטֵר אֶחָד בִּלְבַד
עָמַד עָלֵינוּ וְהַקָּדוֹשׁ
בָּרוּךְ הוּא הִצִּילָנוּ
מִכְּחֻלֵּי הַמַּדִּים.

נָדַרְנוּ לִשְׁמֹר נְגִיעָה עַד סוֹף
הַשָּׁנָה. הֶחֱזַקְתִּי מַעֲמָד אַרְבָּעָה
יָמִים. תַּחַת פֶּסֶל אַבְרָהָם לִינְקוֹלְן
לְיַד הַסִּפְרִיָּה הָעִירוֹנִית בְּרָמַת גַּן
שֻׁחְרַר הָעֶבֶד מִכְּבָלָיו הַתַּחְתּוֹנִים.

יב.

מִגִּבְעַת הַסְּנִיף צָעַדְנוּ בְּתַהֲלוּכָה בִּרְחוֹב יְרוּשָׁלַיִם
בִּבְנֵי בְּרַק, כְּחֻלִּים־לְבָנִים כְּנֶגֶד לוֹבְשֵׁי שְׁחֹרִים.
יְלָדִים מְסֻלְסְלֵי פֵּאוֹת הִבִּיטוּ בָּנוּ סַקְרָנִים.
אַבְרֵכִים צָעֲקוּ כְּנֶגְדֵּנוּ עַזֵּי פָּנִים: מִזְרוֹחְנִיקִים!
מִזְרוֹחְנִיקִים! כִּפָּה סְרוּגָה, מִכְנָסַיִם קְצָרִים,
סֵדֶר בֹּקֶר בִּגְמָרָא, סֵדֶר עֶרֶב מוּל הַטֵּלֵוִיזְיָה,
מְפַצְּחִים גַּרְעִינִים עַל הַבַּרְזִלִּים בְּעֶרֶב שַׁבָּת —
אֵין טַעַם וְאֵין רֵיחַ. חִלּוּל הַשֵּׁם בָּרַבִּים!

רֵיחַ תּוֹרָה עָמַד בֵּינֵינוּ: זֵעָה, בְּעֵרָה וּדְיוֹ.
הַמַּדְרִיךְ הֵרִים אֶת יָדוֹ וַאֲנַחְנוּ שָׁאַגְנוּ דִּבְרֵי הָרַב זָצָ"ל:
"הַצַּדִּיקִים הַטְּהוֹרִים אֵינָם קוֹבְלִים עַל הָרִשְׁעָה
אֶלָּא מוֹסִיפִים צֶדֶק".

הוֹסַפְנוּ לָלֶכֶת עַד הַבָּמָה הַגְּדוֹלָה
בְּרֹב עַם, בְּרַחֲבַת הָעִירִיָּה
צָרַח שְׁלִיחַ הַתְּנוּעָה:
"וְאַתֶּם שֵׁבֶט הָרֹאֶ"ה —
עַמִּיצוּר יִקָּרֵא שִׁמְכֶם!"

זָרַקְנוּ אֶת הַכִּפּוֹת בָּאֲוִיר
כְּכֻמְתּוֹת. אֶת שֶׁלִּי לֹא מָצָאתִי. 

יג.

כְּשֶׁרַבּוּ הַגְּזֵרוֹת נָהַרְנוּ אֶל הַכִּכָּרוֹת
זָעַקְנוּ שִׁיר לַמַּעֲלוֹת, הֵנַפְנוּ שְׁלָטִים, מִמַּעֲמַקֵּי הַקָּהָל
חָתַמְנוּ בְּשִׁירַת "הַתִּקְוָה" וַ"אֲנִי מַאֲמִין".

הַמָּשִׁיחַ שׁוּב הִתְמַהְמֵהַּ בְּצֹמֶת כִּסּוּפִים.
בַּבְּקָרִים חִפַּשְׂנוּ אֶת דְּמוּתוֹ וּדְמוּתֵנוּ
בִּתְמוּנוֹת הַהַפְגָּנָה, בְּשַׁעֲרֵי עִתּוֹנִים.
סְטִיקֵרִים כִּסּוּ אֶת דַּלְתוֹת חֲדָרֵינוּ
לֹא נוֹתַר עֲלֵיהֶן חָלָל פָּנוּי. הֵיטֵב בֵּרַרְנוּ
מִי לָנוּ, מִי לְצָרֵינוּ וּמִי מֵצֵר אֶת אֶרֶץ קָדְשֵׁנוּ.
בְּגוּפֵנוּ עָצַרְנוּ אֶת הַתְּנוּעָה, אָחַזְנוּ יָדַיִם
בְּשַׁרְשֶׁרֶת אֱנוֹשִׁית. הָיִינוּ עַם הַנֶּצַח. לֹא פָּחַדְנוּ
מִדֶּרֶךְ אֲרֻכָּה, מִשְּׁעוֹת נְסִיעָה מִתְאַחֲרוֹת לַעֲצָרוֹת.
הֶאֱמַנּוּ: הָיֹה לֹא תִּהְיֶה. בַּגְּאֻלָּה אֵין נְסִיגוֹת.

בְּשִׁלְהֵי הַהַפְגָּנָה הַגְּדוֹלָה נֶגֶד הָעֲקִירָה
בְּשׁוּלֵי הֶהָמוֹן הַנִּשְׂרָךְ מֵהַכִּכָּר אֶל רְחוֹבוֹת
צְדָדִיִּים, זִהִיתִי לְרֶגַע אֶת פָּנֶיהָ הַטּוֹבוֹת
שֶׁל הַנַּעֲרָה שֶׁלְּיָמִים תִּהְיֶה אִשְׁתִּי
בְּתַחֲנַת אוֹטוֹבּוּס בְּחֻלְצָה כְּתֻמָּה
קוֹרֵאת תְּהִלִּים. הִיא הִבִּיטָה בַּסֵּפֶר,
אֲנִי הִבַּטְתִּי בָּהּ. יָדַעְתִּי שֶׁאַף עַל פִּי
שֶׁתִּתְמַהְמֵהַּ, עִם כָּל זֶה, אֲחַכֶּה לָהּ.

יד.

ג' בֶּאֱלוּל תשפ"א, אֲנִי מְצַלֵּם סֶרֶט בְּהַר הַזֵּיתִים.
מְזַהֶה אֶת גַּג בֵּיתִי בִּדְרוֹם הָעִיר, רָחוֹק
מִגּוּף הָאַגָּנִים הַקְּדוֹשִׁים. אֲנִי נִמְשָׁךְ
אַחַר קוֹל שִׁירַת הַנְּעָרִים,
אַחַר מִזְמוֹרֵיהֶם הַמֻּכָּרִים.
הוֹלְכִים בְּעִקְבֵי הַצֹּאן וּמִצְטוֹפְפִים בֵּין הַקְּבָרִים.
סְבִיב צִיּוּן הָרַב זָצָ"ל שָׁרִים חֲבוּקִים: "בֶּן אָדָם, עֲלֵה
לְמַעְלָה, עֲלֵה".

מִתְנוֹעֲעִים בַּמָּקוֹם. בְּמִשְׁקְפֵי הַשֶּׁמֶשׁ
הַר הַבַּיִת נִשְׁקָף, בּוֹהֵק. חֹם יוֹקֵד עַל כִּפַּת הַזָּהָב,
מַלְהִיט כִּפּוֹת הַתַּלְמִידִים, עֵינֵיהֶם עֲצוּמוֹת
כְּבִתְפִלַּת עֲמִידָה. בִּי לֹא מַשְׁגִּיחִים
גְּלוּי רֹאשׁ, שׁוֹתֵק בְּצַד הַחֲבוּרָה.

שָׁב לְבֵיתִי, פּוֹתֵחַ שׁוּב סֵפֶר "אוֹרוֹת"
בְּפִתְחוֹ הַקְדָּשָׁה מֵהָרַב, מְחַנֵּךְ הַכִּתָּה:
"שֶׁיּוֹסֵף אוֹרְךָ לְאוֹר הָאֻמָּה"

תַּחְתָּיו רָשׁוּם בִּכְתַב יָדִי הַיַּלְדִּי:
"לְהַשֵּׁם הָאָרֶץ וּמְלוֹאָהּ"
וְאַחֲרָיו נִכְתַּב שְׁמִי הַמָּלֵא
כְּגַפְרוּר הַשּׁוֹאֵל אֵשׁ מֵאֲבוּקָה. 

טו.

מַה מֵּעִיר אוֹתִי שׁוּב לִפְנוֹת בֹּקֶר?
אֲנִי מִתְיַשֵּׁב, כּוֹתֵב. גַּם הַלַּיְלָה
צַר לִי לְאַכְזֵב, לְהִתְאַכְזֵב.
מְשׁוֹרֵר הַתְּשׁוּבָה לֹא נוֹלַד
בְּגוּפִי. אֲנִי מַבִּיט בַּתְּמוּנָה
רוֹאֶה אֶת הָרוֹאֶה הַגָּדוֹל
חָשׁ אוֹתוֹ לוֹחֵשׁ בְּאָזְנַי:
"מְקוֹם מְנוּחָתֵנוּ
הוּא רַק בֵּאלֹהִים"

עַכְשָׁו אֲנִי יוֹדֵעַ:
לֹא עָמַדְנוּ עַל טִיבוֹ
וְנִשְׁבַּעְנוּ לָרִיק בִּשְׁמוֹ.
וַאֲנִי יוֹדֵעַ אוֹתוֹ
רַק רָחוֹק
רַק עַכְשָׁו
כַּעֲבֹר שָׁנִים.

מהספר "שירים על סף" שעתיד לראות אור בקרוב בסדרת כבר, הוצאת מוסד ביאליק.
