עליית הימין החדש היא אולי התופעה החשובה ביותר בפוליטיקה המערבית ב–20 השנה האחרונות. נשיאותו של טראמפ והאיום שהיא הציבה על מנגנונים בינלאומיים, הברקזיט, ממשלת הימין הקיצוני באיטליה, מארין לה פן במרחק נגיעה מהנשיאות בצרפת — אלה רק ההשלכות הראשונות של שינוי דרמטי, שאחריתו עדיין לא ידועה. עליית הימין החדש מלווה בשינוי היחס שלו לישראל: אם בעבר הימין הקיצוני תיעב את מדינת היהודים כמו שתיעב יהודים, כעת הוא מתחיל לראות בה שותפה פוטנציאלית, ואפילו מודל. "אלטרנטיבה: ימין חדש לגרמניה?" של אלדד בק ממחיש היטב את המהפך הזה. זהו ספר מטריד — לא בגלל הפרופיל של המפלגה שעומדת במרכזו, אלא בשל הקריאה הטמונה בו לממשלת ישראל לחבק את הימין הקיצוני ולהפוך למעשה לבעלת בריתו בעיצובו של סדר אזורי ועולמי חדש.

מאז מלחמת העולם השנייה, וביתר שאת לאחר איחוד שתי הגרמניות ב–1990, לא היתה מדינה שסימלה את הסדר הליברלי החדש, על מעלותיו ומגרעותיו, יותר מאשר גרמניה. גרמניה שימשה אתגר מיוחד לימין הקיצוני: מפלגות הימין הקיצוני, ובראשן המפלגה הניאו־נאצית הגרמנית (NPD), לא הצליחו להיבחר מעולם לבונדסטאג; רעיונותיהן נחשבו תמיד למוקצים מבחינה פוליטית ועל פעילותן הוטלו מגבלות שונות.

את הטאבו הזה הצליחה לשבור מפלגת "אלטרנטיבה לגרמניה", שנוסדה כמפלגה יורו־סקפטית רדיקלית ב–2013. על רקע ההתנגדות להחלטתה של אנגלה מרקל לפתוח את שערי גרמניה לפליטים מהמזרח התיכון ומאפגניסטאן, נהפכה "אלטרנטיבה" ב–2017 למפלגה השלישית בגודלה בבונדסטאג ולמפלגת האופוזיציה העיקרית. היא שאבה לתוכה את הזרמים הניאו־נאציים בפוליטיקה הגרמנית, אבל גם בורגנים שמרנים שהתנגדו לפרויקט איחוד אירופה, רומנטיקנים של העבר הגרמני, מתנגדי הגירה ממעמד הפועלים במזרח, ואפילו כמה שמאלנים וירוקים לשעבר. בבחירות הכלליות ב–2021 המפלגה נחלשה מעט, אולם על רקע משבר האנרגיה היא שוב מתחזקת בסקרים ובמדינות מחוז מסוימות. ברור שכמו אחיותיה האירופיות האחרות, היא לא עומדת להיעלם.

על "אלטרנטיבה" מוטל חרם מצד יתר המערכת הפוליטית בגרמניה, וגם מפלגות הימין השמרני נמנעות מלהקים עמה קואליציות. הקהילה היהודית חוששת מהמפלגה ומתרחקת מכל מגע איתה; ערב הבחירות הכלליות האחרונות פירסמו 60 ארגונים יהודיים, וביניהם המועצה המרכזית של יהודי גרמניה, קריאה שלא להצביע ל"אלטרנטיבה" בשל "אנטישמיות, גזענות ושנאת אדם". זוהי גם העמדה הרשמית של מדינת ישראל; נבחרים מטעם "אלטרנטיבה" אינם מתקבלים בארץ במעמד רשמי, אלא רק בביקורים פרטיים.

אלדד בק, שליח "ישראל היום" לגרמניה (וקודם לכן שליח "ידיעות אחרונות"), הוא העיתונאי הישראלי שגילה את העניין הגדול ביותר ב"אלטרנטיבה"; ספרו הוא פרי מסע בן שמונה שנים בעקבות המפלגה וראשיה: סיפור צמיחתה, המתחים הפנימיים שמהם היא סובלת והרעיונות הרדיקליים שהיא מבקשת לקדם. הספר מסתיים בקריאה חד־משמעית ללגיטימציה של "אלטרנטיבה". "זמן רב מדי נתנה ישראל לאירופה, ולגרמניה בפרט, לנהוג בה בצורה מעליבה ולעשות בה ככל העולה על רוחה", כותב בק. "השותפים הטבעיים של ישראל כיום לשינוי משוואת היחסים העקומה עם אירופה הם על פי רוב מפלגות 'הימין הרחוק' שמוכנות לשבור את הכלים גם בכל הנוגע לעוינות המסורתית כלפי ישראל. זו הזדמנות שאין לוותר עליה".

סימטריה כוזבת

לזכותו של בק ייאמר שהוא לא מייפה את פניה של "אלטרנטיבה" יותר מדי. הוא דן בתיאוריות הקונספירציה הרווחות בין אוהדיה, מצטט התבטאויות אנטישמיות של אינטלקטואלים ופוליטיקאים שהיו חברים במפלגה או מזוהים איתה, ועוסק בהרחבה בדמותו של ביורן הקה, המנהיג הלא רשמי של "הכנף" — האגף במפלגה שמזוהה עם רביזיוניזם היסטורי ועם הלאומנות הגרמנית הפולקית (תפישה שרואה בעם קהילה אורגנית הקשורה לאדמה ולדם משותף, ושהיתה אחד מיסודות הנאציזם). חברי הכנף מתנגדים לזיכרון ההיסטורי שנכפה לשיטתם על הגרמנים. בין השאר, הם מתעבים את אנדרטת השואה בברלין, שאותה כינה הקה "אנדרטת הבושה"; אחד ממנהיגי הכנף חולל את המיני־שערורייה התורנית בחודש שעבר, כשהצטלם רוקד על האנדרטה. חברי הכנף הם מיעוט במפלגה, אולם יש להם השפעה עצומה על הקו האידיאולוגי שלה ועל זהות יושבי הראש; הקה אף הוכתר בחודש שעבר על ידי "דר שפיגל" כדמות המשפיעה ביותר ב"אלטרנטיבה".

אלא שכל אלה בטלים, לשיטתו של בק, אל מול שתי זכויות גדולות שעומדות ל"אלטרנטיבה": ההתנגדות שלה לאיסלאם, והתמיכה במדיניות הימין הישראלי. "ראשי הקהילות (היהודיות) שבויים ברובם בתפישת הימין הלאומני האירופי כאיום קיומי עבור היהדות", כותב בק, "אך יותר ויותר מחברי וחברות קהילותיהם מזהים את האיום הקיומי במקום אחר: קהילות המהגרים הערביות והמוסלמיות". בק מאשים את השלטונות בגרמניה ואת הארגונים היהודיים בטשטוש האופי האנטישמי של תקיפות שבוצעו נגד יהודים על ידי מוסלמים, רובם מהגרים, ובעיסוק־יתר בימין הקיצוני. את המשפט שנערך לפעיל הימין שטפן באלייט, שניסה לבצע טבח בבית כנסת בהאלה ביום כיפור באוקטובר 2019, מכנה בק "משפט ראווה", שנועד להעמיק את הדה־לגיטימציה של הימין הקיצוני ולטשטש את האלימות המוסלמית.

לשיטתו, "אלטרנטיבה לגרמניה", שמבקשת לאסור הקמת צריחי מסגדים, לגרש שוהים לא חוקיים ולסגור את הגבולות, מבטאת טוב יותר את התחושות והפחדים של יהודי גרמניה היום. כהוכחה, הוא עוסק בהרחבה בהקמתו של תא יהודי זעיר בשורות המפלגה, המורכב כמעט כולו מיוצאי חבר המדינות, שגרמניה קלטה בזרועות פתוחות לאחר נפילתה של ברית המועצות. אם בק מודע לאירוניה, הוא לא מציין זאת.

אין ויכוח על הפגיעות האנטישמיות ביהודים וברכוש יהודי שבוצעו על ידי מוסלמים במערב אירופה בשנים האחרונות (אלימות הימין הקיצוני כלפי מיעוטים זינקה גם היא), אבל הסימטריה שעורך בק בין תקריות אנטישמיות מימין ובקרב מהגרים מפספסת את טיבם האמיתי של לקחי מלחמת העולם השנייה ואת מהות החשש מפני הפשיזם. פשעי הנאצים יכלו להתרחש רק בזכות השילוב בין האידיאולוגיה האנטישמית לכוחה של המדינה המודרנית. האיום הקיומי על יהודים — ועל קהילות נרדפות בכלל — אינו האלימות המזדמנת (קשה וחמורה ככל שתהיה) של שחורים בניו יורק בשנות ה–70 או מהגרים ובניהם בפרברי פריז, אלא הרגע שבו השלטון משתמש בכוחו לפעול נגדם, כשהוא שולח מיליציות חמושות, או מעודד פרעות, או מגרש אותם או רוצח אותם בשיטתיות מפלצתית שרק המדינה מסוגלת לה.

באותו האופן, אין סכנה שהאיסלאם הקיצוני יתפוס את השלטון בגרמניה ובצרפת וישמיד בשיטתיות את מתנגדיו, ואין כיום שום סיוע של מנגנוני המדינות הללו למי שפוגעים ביהודים או ברכוש יהודי; לכל היותר ישנה אכיפה נרפית, שדווקא סופגת ביקורת. אבל יש בהחלט סכנה — ליהודים, ללא יהודים ולשלום העולם — מעלייה של כוחות ריאקציונריים, גזעניים, לאומניים ואנטי־דמוקרטיים ברחבי המערב ומעבר לו. העמדה שמבטאים ראשי הקהילות היהודיות אינה איזה עיקום אף לנוכח התבטאות לא נעימה של פוליטיקאי ימני במרתף בירה — זהו החשש מהפקדתם של מנגנוני המדינה בידיו, חשש שרק מתגבר לנוכח הסעיף במצע של "אלטרנטיבה" הקורא לבטל את המגבלות שהוטלו על גרמניה בתום מלחמת העולם השנייה.

ללמוד מהניסיון הישראלי

עוד יותר מאשר המאבק המשותף במהגרים ובאזרחים מוסלמים, קוסמת לבק האהבה שניצתת בימין הקיצוני החדש כלפי ישראל וכלפי הימין הישראלי. השידוך הזה נולד ממפגש של אינטרסים ותפישות עולם גם יחד: הימין הישראלי והימין הקיצוני ביבשת מתעבים את האיחוד האירופי ומייחלים למפלתו, שניהם בזים ומתנגדים לחוק הבינלאומי ולמנגנוני הבקרה והמשפט שלו; ושניהם חולקים אותה פרנויה מפוליטיקה מגדרית ומפוליטיקת זהויות (מצביעי "אלטרנטיבה" הם ברובם גברים; התמיכה במפלגה היא בעיקר בגילאי 30–60).

יותר מכל, הימין האירופי מזהה בישראל מודל לחיקוי של לאומנות אתנית מסתגרת שמצליחה להיחשב עדיין כדמוקרטיה. הרעיון של מדינת לאום סגורה — לאום של דם ושל דת — שמקצה משאבים בעיקר ליהודים, ששולטת בכוח על מי שאינם יהודים ומסרבת לאזרח אותם, הוא החלום של הימין הקיצוני באירופה. הקשר שפעיליו מבקשים לכונן עם ישראל אינו מבוסס על דחיית העבר הנאצי, אלא על ההווה הלאומני. "ישראל מתמודדת כבר עשרות שנים עם הבעיות שמגיעות אלינו עכשיו, אך אצלנו לא מודים בכך", אומרת יו"ר המפלגה לשעבר, פראוקה פטרי, לבק. "ישראל מיישמת מדיניות לאומית מציאותית", אומר נציג "אלטרנטיבה" בדיון בבונדסטאג בנוגע לסיפוח השטחים.

עוד יותר מאשר המאבק המשותף במהגרים ובאזרחים מוסלמים, קוסמת לבק האהבה שניצתת בימין הקיצוני החדש כלפי ישראל וכלפי הימין הישראלי — שידוך שנולד ממפגש של אינטרסים ותפישות עולם גם יחד

לזהות האידיאולוגית נוסף גם צד מעשי: הימין הקיצוני יכשיר את הכיבוש, וישראל תכשיר את הימין הקיצוני. חיבוק רשמי מישראל לא רק יועיל ל"אלטרנטיבה" בקלפי, אלא גם עשוי לשבור את הטאבו על שיתוף המפלגה בקואליציה בגרמניה. בק עצמו מבחין שפטרי "רצתה מאוד להתראיין לעיתון ישראלי, כדי לשבור מחיצות ודעות קדומות". היא ובעלה אף נסעו לביקור פרטי בהתנחלויות בגדה המערבית; בק מתגאה בהיותו העיתונאי הישראלי הראשון שהעניק לה במה. הוא מחמיא לניסיון של "אלטרנטיבה" להוציא את תנועת ה–BDS מחוץ לחוק, ולתביעתם של בכירי המפלגה להכיר בסיפוח ההתנחלויות ולהעביר את שגרירות גרמניה לירושלים. כשהוא מספר כיצד חומרים שהוא העביר לסגנית יו"ר המפלגה בבונדסטאג הובילו לשאילתה על תמיכת גרמניה באונר"א, כבר לא ברור היכן עובר הקו מפריד בין המחבר למושא המחקר שלו.

לאורך הספר כולו, הראיונות שעורך בק עם ראשי "אלטרנטיבה" והאינטלקטואלים התומכים בה מובאים בצורתם הגולמית, והשאלות שהוא מציג להם ("מה את אומרת למי שטוענים ש'אלטרנטיבה' היא מפלגה נאצית?" או "איזו גרמניה עתידית אתה מאחל לעצמך?") הן מהסוג ההרמות להנחתות שפוליטיקאים אוהבים. גם במסגרת גישה המבקשת להביא את הדברים בשם אומרם היה מקום לתמצת את עיקרי הדברים, ולא להגיש לקורא תמלילי שיחות ונאומים, הנמשכים לעתים על פני יותר מעשרה עמודים, בכתב זעיר, וכוללים גם אמירות סתמיות לחלוטין, החל ב"הגיעה העת לפעול יחדיו ובאופן אחראי" ועד "זו תחילתה של פוליטיקה חדשה".

חיבוקים לפוטין וליאיר נתניהו

הבמה הפתוחה שמעניק בק לראשי "אלטרנטיבה" וההתמקדות שלו ב"נקודה הישראלית" באות על חשבון ניתוח של עליית הימין הקיצוני בהקשרים רחבים יותר. מעבר לנתונים ספורים, אין בספר הרבה מידע על מגמות סוציולוגיות בקרב תומכי המפלגה, ולא דיון במשמעויות העמוקות של רעיונותיה ביחס לאתגרים החדשים העומדים בפני האנושות. מהן ההשלכות של ההתנגדות החריפה של "אלטרנטיבה" לחקיקת האקלים, הן במישור הלאומי והן כחלק מהתנגדותה למנגנונים בינלאומיים? כיצד ישפיע פירוק האיחוד על היכולת לשמור על השלום באירופה, ובעולם בכלל? מי ייקח את התפקיד שממלאים גופים בינלאומיים ובראשם האיחוד ברגולציה טכנולוגית ובהגנה על הפרטיות? אלה שאלות שהספר נמנע מלבחון ברצינות, ואולי לא במקרה.

הוועידה האחרונה של "אלטרנטיבה" התפזרה מוקדם מהמתוכנן בעקבות סירוב הקה ו"הכנף" במפלגה להגדיר את הפלישה הרוסית לאוקראינה כמלחמה; אולם ההערצה של הימין הקיצוני לפוטין — ולפוטיניזם — מוזכרת על ידי בק בחטף בלבד. בחשבון הטוויטר שלו בק מכנה את עצמו "אזרח העולם"; הייתי רוצה לדעת מה חושבים תומכי "הכנף" על הביטוי הזה. לבסוף, חסרה לי גם הערכה רצינית בדבר עתידו של הימין הקיצוני: בינתיים נראה כי האיחוד האירופי צולח את המשברים שעוברים עליו ואף מתחזק בזכותם; ייתכן שבק דורש מישראל להמר על סוס מפסיד.

ספרו של בק מסתיים באזהרה לקובעי המדיניות הישראלים, שדוחים את "היד המושטת" מצדה של "אלטרנטיבה": "חלון ההזדמנויות הזה לא יישאר פתוח לנצח. סופה של דחייה להיענות בדחייה". לצערי, החשש שלו מוגזם. הגלישה הישראלית לאפרטהייד מגבירה את החיכוך עם הכוחות הליברליים בעולם — ואת הפיתוי שבחיבוק עם הימין הלאומני, שמצדו ממשיך בחיזוריו אחרי ישראל. סדרת ציוצים אלימה של יאיר נתניהו נגד האיחוד האירופי ("הארגון הגלובליסטי המרושע") כבר הפכה אותו לגיבור בעיני "אלטרנטיבה"; עלייתו של הכהניזם וההיסחפות של הליכוד ימינה מבטיחות כי בסדר העולמי החדש שמדמיין הימין האירופי, ימשיך להישמר לנו מקום של כבוד.
