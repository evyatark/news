אָסָף סֶרְגֵּיֵבִיץ' הִתְאַלְמֵן

הַטּוֹב הָאֱנוֹשִׁי חוֹמֵק

מֵהָעוֹלָם. נִשְׁאֶרֶת רַק שָׂפָה.

עוֹלָם חָדָשׁ עַצְמוֹ כָּפָה,

עַל סֵדֶר־יוֹם, עַל הַשִּׁגְרָה,

זִיפִים, מִיּוֹם שֶׁנִּקְבְּרָה

צָמְחוּ פִּרְאִי.

יֵשׁ חֲזָרָה אֶל הַטִּבְעִי —

הִיא יַבָּשָׁה וְהוּא כְּאִי.

אֵין כָּאן מוֹטִיב חָדָשׁ חָלִילָה,

הִיא בְּמוֹתָהּ עָלָיו הִפִּילָה:

תִּחְקוּר יוֹמִי אֵצֶל אִמָּהּ,

סַדְנַת זוּגִיּוּת — לֹא הֻשְׁלְמָה,

טִפּוּל בִּילָדִים. שְׁלוֹשָׁה.

מָנוּי חָדְשִׁי לְ"לְאִשָּׁה".

לִהְיוֹת אַלְמָן בְּגִיל שְׁלוֹשִׁים

סְבִיבוֹ הֵחֵלּוּ רְחָשִׁים,

שֶׁל שַׁדְכָנִיּוֹת לְעֵת מְצֹא,

עֲמִיתוֹתָיו הַנִּמְרָצוֹת,

עָטוּ עָלָיו בְּלִי הַפְסָקָה.

הָיְתָה וַדַּאי עוֹלָה תְּפוּקָה

אִם כָּךְ הָיוּ עוֹבְדוֹת תָּמִיד.

הוּא לְפָנָיו רָאָה עָתִיד

שֶׁל זִכָּרוֹן.

בְּגִיל שְׁלוֹשִׁים וּכְבָר אַלְמָן

גּוּפָהּ בָּאֲדָמָה נִטְמַן.

וַחֲלוֹמָהּ נִטְמַן עִמָּהּ:

לִחְיוֹת בְּלִי לְהַרְגִּישׁ אַשְׁמָה,

סַדְנַת זוּגִיּוּת. לֹא הֻשְׁלְמָה.

לִהְיוֹת לְבַד. בַּבַּיִת דְּמָמָה.

כָּל יוֹם הוֹסִיף לָרְשִׁימָה

"אָה... בַּיִת צְמוּד קַרְקַע עִם גִּנָּה".

שֶׁל חֲלוֹמוֹת שֶׁלֹּא מִמְּשָׁה,

הַדַּף בַּאֲרוֹנִית קְדוֹשָׁה

נִשְׁמַר מֵעַל כִּיּוֹר סָדוּק.

שָׁלָב ב' כְּמִתְאַבֵּל אָדוּק

הָיָה לִבְנוֹת תָּכְנִית.

שְׁלַבִּים לִפְעֻלָּה,

תַּפְנִית

בָּעֲלִילָה:

לְבִצּוּעָהּ חִכָּה עַד הַשָּׁעָה

שֶׁבָּהּ יָכוֹל לִזְכּוֹת בְּכַפָּרַת הַהֲנָאָה

בַּיְּצִיאָה מֵהָאֲבֵלוּת.

הָיָה זֶה תֹּם עִדָּן שֶׁל הַגָּלוּת,

עֲבוּר הוֹרָיו. יוֹם לֵדָתוֹ.

צַבָּר רִאשׁוֹן מִיַּנְקוּתוֹ,

מֵאֲצֻלַּת הָעֲלִיָּה,

בֶּן אֲסִירַת צִיּוֹן גֵּאָה,

בְּחוּג שַׁחְמָט, מַתְנָ"ס בֵּית אֵל,

אִבֵּד אֶת הָאֱמוּנָה בָּאֵל,

אִם כִּי שָׁמַר עַל קֶשֶׁר מֵרָחוֹק,

רֵיחַ חָלוֹת כְּזִכָּרוֹן מָתוֹק,

וּמְתוּקָה גַּם הָאַחְוָה

שֶׁל חֲבֵרָיו לַיְּשִׁיבָה.

אַךְ טַעְמָהּ שֶׁל

הַנְּקָמָה

שֶׁנִּרְקְמָה

כָּעֵת, מָתוֹק יוֹתֵר.

תְּחִלָּה עָלָיו לְהִסְתַּתֵּר,

לְהֵעָלֵם, לְהִתְפַּטֵּר.

חֶשְׁבּוֹן מִפֶיְסְבּוּק לְהָסִיר,

אֶת כָּל חַיָּיו אִרְגּוּן סְבִיב צִיר

אֶחָד. וּמַטָּרָה סְדוּרָה,

מַצָּב הִצְרִיךְ שְׁבִירַת צוּרָה.

אֵצֶל הַסָּבְתָא הַיְּלָדִים,

הוּא הִתְעַלֵּם מֵעוֹרְכֵי דִּין,

דָּרְשׁוּ לָלֶכֶת לְמִשְׁפָּט!

מַמָּשׁ הָיָה לָהֶם אִכְפַּת

מֵחֲלוֹמוֹת שֶׁהִתְנַפְּצוּ,

אֵין דִּין לְכָךְ וְאֵין פִּצּוּי.

כַּמָּה בָּנָאלִי הָאָבְדָן,

אַךְ רַע יוֹתֵר לִהְיוֹת קָרְבָּן

שֶׁל סְעִיפִים, הִסְתַּיְּגוּיוֹת

וּשְׁאָר לָקוּנוֹת חֻקָּתִיּוֹת.

הוּא כְּמוֹ בְּגִיל הַנְּעוּרִים

שׁוּב הִתְקַיֵּם מִנְּדוּדִים.

אָבִיו הָיָה צַיָּר מֻכָּר בִּירוּשָׁלַיִם שֶׁל הַתִּשְׁעִים

מֵאָז שֶׁנִּפְגַּע בְּאַחַד הַפִּגּוּעִים,

נִשְׂרַף אֶצְלוֹ אֵיזֶה נָתִיךְ,

שָׁקַע בְּרִשּׁוּמֵי טְרִיפְּטִיךְ

תְּקוּמָה.

תּוֹרָה, נָבִיא וַאֲדָמָה.

בְּתוֹר נָבִיא — מְנַחֵם בֶּגִין,

נוֹשֵׂא נְאוּם, אוֹחֵז בַּדֶּגֶל.

הוֹרִים פְּרוּדִים — כְּתָמִים שֶׁל דְּיוֹ,

בְּנוֹ לָן בַּסְּטוּדְיוֹ אֶצְלוֹ,

בְּצֵל אָבִיו הַבּוֹהֶמְיָן.

רַק בְּמַזָּל נִהְיָה שַׁתְיָן

וְלֹא מָכוּר... מֵאָז עָזַב אֶת בֵּית הָאֵל,

אֵינוֹ מֵשִׁיב וְרַק שׁוֹאֵל.

חַבְרֵי אָבִיו מֵהַבִּירָה

תָּמְכוּ בּוֹ אָז בִּנְעוּרָיו:

מַרְצֶה בַּלְשָׁן, אֵשֶׁת חֶבְרָה,

עֵזֶר תָּמִיד לְכָל צָרָה.

גֵ'יְמְס הַסּוּדָנִי, מְשׁוֹרֵר צָעִיר

וְגַם אַחַת מִחוּץ לָעִיר.

בֵּית הַטּוֹרֵף בְּצַהֲלָה

הָיָה מֻקָּף חוֹמָה גְּדוֹלָה

וּמַצְלֵמוֹת הָאַבְטָחָה

קִבְּלוּ פְּנֵי זָר בְּלֹא בְּרָכָה.

שָׁלָב רִאשׁוֹן. לִפְעֹל מִיָּד,

בִּיחִידַת דִּיּוּר עַל יַד,

לִשְׂכֹּר חַדְרוֹן קָטָן וְצַר

בּוֹ הַמַּרְצֶה נִהְיָה דַּיָּר.

אֲוִיר שָׂנוּא אָמְנָם נָשַׁם,

אַךְ בְּפִנְקַס הַכֹּל רָשַׁם.

הָעִצּוּרִים וְהַתְּנוּעוֹת

הֶחֱלִיף בְּחֵקֶר עֲצִירוֹת

וְהַתְּנוּעָה

שֶׁל הַיְּבוּאָן מִצַּהֲלָה.

מָתַי יוֹצֵא הוּא מִבֵּיתוֹ,

מָתַי בָּאִים גַּם אֶל אִשְׁתּוֹ:

גַּנָּן, עוֹזֶרֶת, מְנַקָּה

בְּכָל שָׁעָה, בְּכָל דַּקָּה.

מָתַי סִיאֶסְטָה, מְנוּחָה,

מָתַי הִיא מַשְׁמִיעָה צְרָחָה.

וּבֵין חָמֵשׁ לְשֶׁבַע — חֹר.

יוֹצֵאת תָּמִיד בְּגִ'יפּ שָׁחֹר,

אֶל עֵבֶר יַעַד מִסְתּוֹרִי.

אָסָף נִזְקָק לַאֲחֵרִים.

אָבִיו נִשְׁמָע הַסְסָנִי,

"רַק אֶת אִמְּךָ אַל תְּעָרֵב..."

"הִיא לֹא תֵּדַע" הוּא הִתְחַיֵּב.

אָבִיו דִּבֵּר לְאַט, רָגוּעַ:

"מוּל הֶחָזָק הַסּוֹף יָדוּעַ".

"אֲנִי כְּבָר אַחֲרֵי הַסּוֹף"

עָנָה מִיָּד מִבְּלִי לַחֲשֹׁב.

"תֵּן לִי שָׁבוּעַ לְהַחְלִיט"

חִבֵּק, מָזַג לוֹ עוֹד כּוֹסִית.

שָׁלוֹשׁ כּוֹסִיּוֹת הִסְפִּיקוּ לָהּ,

"תִּרְאֶה, בָּחוּר, זוֹ הַרְעָלָה"

אָמַר הַדּוֹקְטוֹר: "מֶתָנוֹל.

אָחוּז קָטָן, נֶזֶק גָּדוֹל".

הֵישִׁיר מַבָּט, קוֹלוֹ נִנּוֹחַ,

יָרְדוּ שְׁלָגִים, כַּדּוּר הַבְּדֹלַח,

מֵאֲחוֹרֵי גַּבּוֹ נִצְנֵץ

כִּמְרַמֵּז כְּבָר עַל הַקֵּץ...

כָּעֵת חִשֵּׁב כָּל פְּרָט וּפְרָט.

כַּמָּה סָבַל בְּחוּג שַחְמָט,

כָּךְ קַל הָיָה לוֹ בַּלֵּילוֹת,

לִשְׁקֹל אֶת שְׁלַל הַשְּׁאֵלוֹת.

וּכְבָר בְּאֶמְצַע הַשִּׁבְעָה

פִּתְאוֹם נִתְקַל בְּמוֹדָעָה:

"שִׁמְעוּ נָא אֵיזוֹ חֲדָשָׁה!",

נִכְתַּב בְּטוּר בְּ"לָאִשָּׁה",

"מִי זֶה נִתְפַּס בַּעֲדָשָׁה?

עִם כַּלָּתוֹ הַחֲדָשָׁה?

הַאִם נִקְבַּע אוֹ לֹא עֲדַיִן

הַיּוֹם בּוֹ יִשְּׁפֵךְ הַיַּיִן,

בַּחֲתֻנָּה עִם הַיּוֹרֶשֶׁת,

מַשְפִּיעָנִית, כּוֹכֶבֶת רֶשֶׁת.

בַּת הַיְּבוּאָן הַמְּפֻרְסָם".

הַנְּקָמָה הִיא סוּג שֶׁל סַם.

אֵשֶׁת חֶבְרַת הַוָּתִיקָה

לָבְשָׁה אַדֶּרֶת מְפִיקָה

וְאֶל הַבַּיִת בְּצַהֲלָה

נָשְׂאָה רַגְלֶיהָ צָהֳלָה.

גֵ'יְמְס הַפָּלִיט מִדְּרוֹם סוּדָן

אָרַב לְמוֹטִי הַגַּנָּן,

נִדְנֵד, הִצִּיעַ וְחִזֵּר

עַד שֶׁנִּהְיֶה לוֹ לְעוֹזֵר.

וְכָךְ אֶל לֵב הָאֲחֻזָּה,

נִכְנַס, שֻׁתָּף לְהַמְחָזָה

שֶׁל כָּל מָה שֶׁעָתִיד לִקְרוֹת.

אַחֲרֵי נִתּוּחַ שֶׁל שְׁגָרוֹת,

הוּפְקָה מִיָּד תָּכְנִית אֵרוּעַ

הַזּוּג הֶחֱלִיטוּ עַל אֵרוּחַ.

צָנוּעַ. אֶלֶף מֻזְמָנִים.

לֹא שִׁעֲרוּ שֶׁהַזְּמַנִּים

יַכּוּ בָּהֶם בְּלִי הַכָּרָה,

בְּהַנָּחָתָם אֶת הָעֲטָרָה.

הַכֹּל עָמַד. וְקַל וָחֹמֶר,

שִׁלְהֵי אָבִיב, בְּצֵאת הָעֹמֶר,

בַּתַּאֲרִיךְ הַכֹּה נִכְסָף

כְּבָר נֶאֱסַר וְנֶאֱכַף

שָׂרִים מִתּוֹךְ הַמַּקְהֵלָה

שֶׁל יְלָדִים מִגַּן "חֶמְדָּה",

קִטְעֵי קְרִיאָה שֶׁל הַגָּדָה,

כָּל הַקָּהָל הֻקְסַם וְשָׁר,

אַךְ אָז לְפֶתַע נֶעֱצַר

הָרֶצֶף. בְּדִלּוּג שֶׁל עֲרִיכָה,

הָיוּ שְׁנִיּוֹת שֶׁל הַבְלָחָה:

אִשָּׁה זָרָה "מָה נִשְׁתַּנָּה?"

פִּזְּמָה, כְּאִלּוּ שִׁנְּנָה

אֶת הַמִּלִּים הַרְבֵּה לִפְנֵי,

מֵאָז יְמֵי הַ"הוֹפָּה הֵי"

כְּבָר הִתְגַּבְּרָה עַל הַמִּכְשׁוֹל

וּבְבִטָּחוֹן יָדְעָה לִשְׁאֹל

אֶת הַקֻּשְׁיוֹת לִפְנֵי כֻּלָּם.

שֶׁקֶט נִהְיֶה בְּתוֹךְ אוּלָם

הַגַּן.

מָה פֵּשֶׁר שֶׁל אוֹתָהּ אִשָּׁה יָפָה?

אַךְ הַתְּמוּנָה שׁוּב הִתְחַלְּפָה

וְתֶכֶף הָעִנְיָן נִשְׁכַּח.

רַק הַשִּׁירוֹן נִשְׁאַר מֻנָּח.

הָאָב הֵחֵל לָזוּז בְּחֹסֶר נַחַת,

רֵיקָה עָמְדָה הִיא הַצַּלַּחַת,

הִכִּיר הֵיטֵב אֶת הַנַּעֲרָה,

בִּפְנִים עָמְדָה הַסְּעָרָה

לִפְרֹץ.

מַרְאֶה שֶׁלָּהּ כְּמוֹ אֵיזֶה קוֹץ

נִתְקַע. "אֵין זֶה מִקְרֶה"

מִיָּד הֵבִין. "מָה עוֹד יִקְרֶה?"

חָשַׁב. יֵשׁ לַעֲצֹר אֶת הַמִּשְׁדָּר,

עוֹד יְשַׁלֵּם מִי שֶׁחָדַר

לְכָאן!

אֵין לְתָאֵר אֶת הַחֻצְפָּה!

אַךְ כְּבָר הֵחֵלָּה הַחֻפָּה!

וְצִחְקוּקֵי הָרַב

הִפְסִיקוּ דַּאֲגַת הָאָב.

אָחַז בַּבַּת,

נָתַן מַבָּט

מָלֵא הִשְׁתָּאוּת

"וַדַּאי טָעוּת!"

הָיְתָה. בִּלְבּוּל וּמְאַחֵז עֵינַיִם,

אַךְ אָז בִּמְקוֹם: "אִם אֶשְׁכָּחֲךָ יְרוּשָׁלַיִם!"

מִלְמֵל מִשְׁפָּט שֶׁלֹּא קַיָּם

הָרַב: "אִם אֶשְׁכַּח אוֹתָךְ מִרְיָם!"

חִיֵּב לוֹמַר אֶת הֶחָתָן בְּרֶגַע שֶׁל שְׁבִירַת הַכּוֹס.

"עֵת לְהַבְלִיג וְעֵת לִכְעֹס!"

חִשֵּׁב אֶת צְעָדָיו הָאָב

וְאָז בְּקוֹל תִּקֵּן אֶת הָרַב:

"יְרוּשָׁלַיִם!" הוּא הֵעִיר,

"חַיֵּי אָדָם קוֹדְמִים לָעִיר!"

עָנָה הָרַב. חִיּוּךְ כּוֹבֵשׁ,

בִּפְנִים אָמְנָם קְצָת הִתְבַּיֵּשׁ

עַל חֲרִיגָה מֵהַמִּנְהָג,

אַךְ בְּכָל עֵת, תָּמִיד נָהַג

עַל פִּי כְּלָלֵי מַצְפּוּן וְלֵב.

אֱמֶת, תְּחִלָּה לֹא הִתְלַהֵב

כַּאֲשֶׁר אָסָף אֵלָיו פָּנָה,

סָגַר מִזְּמַן אֶת הַחֲתֻנָּה,

מַכָּר שֶׁל בְּנֵי הַמִּשְׁפָּחָה.

אַךְ לְפָנָיו הַהוֹכָחָה

הֻצְּגָה. בְּתוֹךְ הַבַּיִת עָבְדָה,

לְמַטָּה — הַמַּעְבָּדָה.

בְּתוֹךְ מַרְתֵּף הָרְפָאִים,

נִמְזַג לוֹ כֹּהַל רְפוּאִי

אֶל בַּקְבּוּקִים מִמּוּתָגִים,

בְּלִי כָּל סְפֵקוֹת אוֹ סְיָגִים,

הַכֹּל תֹּעַד בְּמַעֲקָב

וּלְבַסּוֹף הִסְכִּים הָרַב

קְצָת לְתַבֵּל רֶגַע הַשִּׂיא,

וּלְהוֹסִיף עִדּוּן טִקְסִי

הַהִתְרַגְּשׁוּת פָּגָה מִזְּמַן

וְהִנֵּה צָץ אוֹתוֹ אַלְמָן..

"חַיֵּי אָדָם קוֹדְמִים..." חָתַךְ,

נָטַל דְּבָרָיו. מִיָּד הָלַךְ.

בְּלִי לְקַבֵּל אֶת הַשָּׂכָר,

בִּמְקוֹם זֹאת, בְּגָרוֹן נִחַר

צָרַח הָאָב בְּעִקְבוֹתָיו:

"אַתָּה גָּמוּר!"

אַךְ הַשִּׁדּוּר

הִמְשִׁיךְ בְּתִעוּדוֹ,

הַיְּבוּאָן אָץ לְחַדְרוֹ,

מֵהֲמֻלַּת הַחֲתֻנָּה.

אֲשֶׁר הִמְשִׁיכָה בַּגִּנָּה.

וְהַצַּלָּם בְּעִקְּבוֹתָיו,

אֶל הַסָּלוֹן נִכְנַס הָאָב,

מֵסִיר הַלּוֹט מֵעַל תְּמוּנָה,

זוֹ הֻבְטְחָה כְּמַתָּנָה

לַזּוּג מֵאֵת צַיָּר מֻכָּר,

אֲבָל מַכָּה אַחַר מַכָּה,

טְרִיפְּטִיךְ — בּוֹ יְלָדִים שְׁלוֹשָׁה,

בְּאֶמְצַע פְּנֵי אוֹתָהּ אִשָּׁה,

הִיא מַחֲזִיקָה אֶת הַתִּינוֹק,

בְּלֹא מֵשִׂים פּוֹרֵץ לוֹ צְחוֹק,

מִבֵּין שִׂפְתֵי הַיְּבוּאָן,

קוֹדֵחַ הוּא חוֹזֵר לַגַּן,

שָׁבוּר, צוֹרֵחַ, מְבֻלְבָּל,

מַפְסִיק הַכֹּל, מְבַקֵּשׁ: חֲדַל!

בִּפְנִים רוֹעֵד, פָּנָיו פְּנֵי מֵת

אֲבָל עַל הַמָּסָךְ כָּעֵת,

הוֹפִיעוּ שׁוּב פְּנֵי הַמֵּתָה,

דְּמָמָה. הַפַּעַם בֶּאֱמֶת הַכֹּל נִקְטַע

וְרֶצֶף טֶקְסְט עַל הַמִּרְקָע —

רֶצֶף חַיִּים אֲשֶׁר נִגְדַּע

תֻּמְצַת בִּסְלִיל הַמִּשְׁאָלוֹת:

בְּלִי לְהַרְגִּישׁ אַשְׁמָה לִחְיוֹת,

לִרְאוֹת שְׁלָגִים וְגֵיאָיוֹת,

לִרְעֹד מְאֻשָּׁר וְלִבְכּוֹת

מֵהִתְרַגְּשׁוּת וּמִשִּׂמְחָה.

הַאִם הַיְּבוּאָן בָּכָה?

סָפֵק אִם כָּךְ בֶּאֱמֶת קָרָה...

אָסָף דִּבֵּר אִתִּי אֶתְמוֹל, נִפְגַּשְׁנוּ עִם הַיְּלָדִים, מֵאָז הַמָּוֶת הוּא רָזֶה, סִפֵּר לִי כָּל מָה שֶׁהָזָה בְּדִמְיוֹנוֹ, סַהֲרוּרִי. אֲנִי חָשַׁבְתִּי בֶּאֱמֶת עַל כָּל הַטּוֹב אֲשֶׁר חוֹמֵק מֵהָעוֹלָם. זֶה לְזִכְרֵךְ, אוֹחְתִי מִרְיָם.
