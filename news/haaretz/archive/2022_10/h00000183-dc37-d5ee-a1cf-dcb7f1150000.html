בכניסה למסיבת הטכנו הגדולה בראשון לציון הצלחתי להבין את פורצי הגדרות

אירועי האלימות ופריצת הגדרות בכניסה למסיבות ענק בידי בליינים עלו לכותרות בשבועות האחרונים. ערב שלם במחיצת מאבטחי מועדון "דזרט" מעלה את השאלה - האם התנאים הבלתי נסבלים בכניסה הם לא אלו שמתדלקים את האלימות?

פריצת הגדרות והאלימות באירועי מוזיקת טכנו עלתה לכותרות ב"זאמנה" בחוות רונית בתחילת ספטמבר. התופעה המשיכה להדהד השבוע עם מעצר 22 חשודים בידי המשטרה בזמן פסטיבל "DGTL" בפארק הירקון אחרי שעשרות ניסו לפרוץ אל המתחם בלי כרטיסים. בעקבות אירועים כאלה בוטלו מסיבות רבות שתוכננו לתקופה האחרונה כמו פסטיבל "דיי זירו" במצדה, אבל בכל זאת נשארו כמה אירועים שאפשר ללכת אליהם כדי להבין איך נראים אירועי טכנו המוניים מבפנים.

בחמישי בלילה נערכה מסיבת ענק במועדון הפתוח "דזרט" בראשון לציון, בו הדיג'יי האוקראינית מיס מוניק (שידועה בשיער הירוק שלה) תיקלטה בחיזוק כמה מהשמות הגדולים מהארץ כמו הצמד Made in TLV והדיג'יי Chaim.

  • באו ליהנות מהמוזיקה, יצאו פצועים וחבולים

הדרך למועדון ארוכה. ה"דזרט" ממוקם בחולות ראשון, צפונית למתחם יס פלאנט והסופרלנד, באיזור שהגישה אליו מורכבת והחניה המוסדרת בו מעטה עד לא קיימת. פקחי השיטור העירוני בראשון חוסמים את כביש הגישה למתחם, ולמבלים יש שתי אפשרויות: האחת היא להגיע מוקדם ולחנות בפנים, והשניה היא להשאיר את האוטו בחולות ומשם ללכת ברגל כ–800 מטר או לקחת שאטל בעשרה שקלים. "זה כמו יציאת מצרים, אבל זה ככה תמיד כי אין שם מקום לחנות" אומרת רותם בדרך לכניסה, צועדת בנחישות על עקבים גבוהים כשהיא מלווה בחברתה שני.

לא ברור מאיפה הקהל אמור לצאת החוצה. אם יקרה כאן משהו, אני חושבת לעצמי, היציאה תהפוך לזירת דריסה אנושית.

כבר בדרך ראיתי שני מקרים של כמעט־התפוצצות: אחד של בליין שקילל את פקחי השיטור העירוני של ראשון לציון שחסמו את המעבר. הפקחים הגיבו בקור רוח, כנראה שזו באמת לא הפעם הראשונה שהם נתקלים בבלגן בכניסה למתחם. המקרה השני היה כבר ממש קרוב לכניסה, כמה נערים יצאו מרכב אחרי שכנראה חנו מוקדם בחניה ובשאר הזמן שתו שם אלכוהול, ואחד מהם התעצבן מאוד. "ירין הבן זונה, זה הכל באשמתו. כל פעם זה ככה ונשבר לי מזה", שמעתי אותו צועק מרחוק, כשחברים שלו מנסים להרגיע אותו. האם זה מה שקורה כשצעיר מתודלק וכועס מגיע לגדרות ולבירוקרטיה של כניסה אחרי שהלך 800 מטר ולא מצא חניה מוסדרת?

בזמן ההליכה, עוד לפני הכניסה, הסאונד שבוקע ממערכת ההגברה הענקית והמצוינת של ה"דזרט" מחלחל לכל עצם בגוף ומושך את המבלים אליו. הקיק, התוף שמתווה את הקצב, מורגש בכל התאים. הוא משנה את קצב הלב לקצב שלו — בום, בום, בום, בום. בגוף של כל מי ששומע את זה, גם אם הוא שונא את הסגנון המוזיקלי הספציפי, קורה דבר אחד: אתה רוצה להיות אחד עם הסאונד הזה. לטבוע בתוכו, להשתכר בתוכו, ללכת לאיבוד ולשכוח הכל. לרקוד שעות ארוכות, להתמסר לחוויה של טראנס, של שחרור אמיתי. וכשבדרך בינך לבין הצלילים האלה עומדת גדר, פתאום גם אני רציתי רק לפרוץ אותה ולהיכנס.

הכניסה למתחם מחולקת לשתיים, כניסה אחת "רגילה" וכניסה אחת למי שהזמינו מראש שולחן לכמה אנשים. יותר מ–20 אנשי צוות אמונים על החלק הזה, שמורכב מבידוק בטחוני קפדני על־ידי מאבטחים, ומבדיקה של תעודות זהות ושל הכרטיס על־ידי אחת העובדות. לנקודת החיכוך הזאת מתנקזים הבליינים שכבר שיכורים מאלכוהול ומהצלילים שבוקעים מהמתחם. הצוות משתדל להכניס את רוכשי הכרטיסים כמה שיותר מהר, או לאפשר למי שהגיע בלי כרטיס לרכוש אחד, אבל המוזיקה כבר מכריחה את כולם לדבר בצעקות והקהל להוט להיכנס פנימה. כל אי הסכמה קטנה עם מארחת או מאבטח הופכת לריב בצעקות בגלל עוצמת המוזיקה, וכשהקהל שזורם מדרך הגישה הולך ומתאסף נהיה צפוף ולוחץ. אבל ניכר שמתחם ה"דזרט" מוכן לאירועים כאלה — המתחם מגודר כדרך קבע ולא בגדר ארעית, והכניסות בו מסודרות, בנויות וממוסדות. זה ההבדל, אומר אחד המאבטחים במקום, בין קיום מסיבה במקום שמותאם לכך לבין "אירוע חוץ" שמצריך גידור ארעי והקמה מאפס של המתחם. במקומות כאלה הגדרות רעועות יותר, ובדרך כלל הכניסות קטנות יותר, מה שמגדיל עליהן את הלחץ.

ערב של יציאה נהיה כבד כספית, אז הם רוצים פשוט להכנס בחינם. הם מאשימים את היח"צנים ולא מבינים שגם אצלנו העלו מחירים

און שלום, מפיק המסיבה

אחרי מתחם הבידוק הראשוני הרוקדים מובלים דרך מאבטח נוסף אל הרחבה הראשית האדירה, שמחולקת למפלס כללי ולמפלס VIP. בתוך המתחם הענק והפתוח יש כבר אלפי בליינים. צעירים ומבוגרים, חלקם אנשי טכנו ותיקים וחלקם בסך הכל רצו לבלות בחמישי בערב של חול המועד סוכות. המקום צפוף, מתחמי הבר עמוסים ועל השולחנות המוזמנים בקבוקי האלכוהול כבר פתוחים ונשפכים. יש עוד מאבטחים בפנים, בפרט במקומות מועדים לפורענות כמו למשל בסמוך לרקדנית שהוצבה בתוך בריכת נוי מרוקנת. כל פעם שמבלה מתקרב לרקדנית ומנסה לדבר איתה המאבטח מייד מרחיק אותו ממנה. בסך הכל 120 אנשי צוות עובדים באירוע, אבל מרוב נסיון לווסת את זרימת הקהל פנימה, לא ברור מאיפה הקהל אמור לצאת החוצה. אם יקרה כאן משהו, אני חושבת לעצמי, היציאה תהפוך לזירת דריסה אנושית.

בשעה 01:30 עדיין יש חבורות שעומדות ומחכות מחוץ לכניסה לשולחנות מוזמנים. "תכתבי שהיח"צנים חרא ואנחנו מחכים כאן כבר מעל שעה", אומרים ירדן והחבורה שלו, שטוענים שהזמינו שולחן לכולם וכעת לא נותנים להם להיכנס. "אנחנו רוצים שולחן, לא סתם כרטיס. כל הכיף זה לשבת ביחד ולהזמין בקבוק". כשאני חוזרת למקום אחרי 20 דקות הם כבר לא שם, אבל פעם נוספת אפשר לראות את אותם חומרים שמהם עשויה האלימות במקומות כאלו: צעירים להוטים לרקוד, לשתות ולצלם סלפיז שנתקלים במכשול שמרחיק אותם מזה, לא משנה כמה הוא לגיטימי.

עדי כחלון ואון משה מפיקי המסיבה שפועלת תחת הליין "טכנו־לוג'י", מסבירים שמנקודת מבטם פריצת הגדרות היא אירוע כלכלי, ונובעת מזה שהאינפלציה השפיעה גם על הבליינים שכבר לא נשאר להם כסף. "ערב של יציאה נהיה כבד כספית, אז הם רוצים פשוט להכנס בחינם. הם מאשימים את היח"צנים (שאחראים על מכירת הכרטיסים ושמירת שולחנות, ש"נ) ולא מבינים שגם אצלנו העלו מחירים. דיג'ייז שלפני שנה היו לוקחים אלפיים שקל כבר דורשים 13 אלף, המשטרה דורשת המון כסף על אבטחת אירוע", אומר משה, ומוסיף שהוא שמע מכמה דיג'ייז מחו"ל שהם כבר לא רוצים להגיע לישראל מחשש ששמם יידבק באירועי אלימות שכאלה. בין השורות כחלון ומשה מתארים משהו נוסף שקרה בסצינה בעקבות המשבר הכלכלי — התפוררות הסולידריות בין המבלים, למארגנים, לדיג'ייז. הם מספרים שכל אחד מנסה לשמור על עצמו ו"לא להידפק", וכשאלכוהול והרבה כסף נכנסים למשוואה, פיצוצים אלימים הם רק שאלה של זמן.

"זה הקיץ הכי קשה שהיה לי בעסק" אומר משה, אבל למרות הכל הוא וכחלון מתכננים הפקה גדולה נוספת בתחילת נובמבר, הפעם לא במגרש הביתי שהוא מועדון ה"דזרט" המצויד והממוסד, אלא בהאנגר 11 בתל אביב. "חייבים לקחת סיכונים, זה הביזנס", הוא אומר, "אנשים חושבים שעושים מזה המון כסף אבל זה לא נכון, הכיף שלי זה לראות אנשים מבלים". ברגע שעוברים את הכניסה, הנקודה שמועדת לפורענות, ברור על מה המפיקים מדברים. בתוך הסאונד של ה"דזרט" באמת אפשר לטבוע, ותוך דקות בודדות כל מה שקרה בדרך למועדון נשכח. כשה"קיק" נכנס, וכשהדיג'יי מנגן את השיר הנכון ברגע הנכון, כל האנרגיה של 4,000 המבלים במתחם משתחררת בבת אחת ברגע שקשה לתאר את העוצמה שלו, אבל קל לתאר איך הוא יכול לאבד שליטה ולהיות אגרסיבי. מיס מוניק כתבה על הערב הזה באינסטגרם שלה: "ראשון לציון, אתם יודעים איך לחגוג".
