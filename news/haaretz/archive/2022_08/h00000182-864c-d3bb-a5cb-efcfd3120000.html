האנדרדוגיות שזכו והגיבור הלאומי שהפך לטראגי: סיכום אולימפיאדת השחמט

הלילה, בעיר צ'נאיי שבהודו, הגיעה לסיומה אולימפיאדת השחמט הגדולה ביותר אי פעם. שתי הנבחרות הזוכות, אוזבקיסטן בגברים ואוקראינה בנשים, לא דורגו בין הראשונות על הנייר. המארחת אמנם הסתפקה בארד בשני המגדרים, אך ההתלהבות הלאומית מהענף במדינה הרקיעה שחקים

טקס הנעילה של אולימפיאדת השחמט, הלילה בהודו. למרות היעדרותה של סין והחרמתן של רוסיה ובלארוס, היתה זו האולימפיאדה הגדולה ביותר עד כהצילום: ARUN SANKAR - AFP

אולימפיאדת השחמט ה-44 ננעלה הלילה (בין שלישי לרביעי) בעיר צ'נאיי שבהודו, לאחר 12 ימי קרבות מותחים שאחריהם עקבו מיליונים ברחבי תבל. סין, אלופת שני המגדרים באולימפיאדה הקודמת - באטומי 2018 (ב-2020 בוטלה האולימפיאדה) - נעדרה בשל הנחיות הקורונה, רוסיה ובלארוס הוחרמו; למרות זאת, היתה זו האולימפיאדה הגדולה ביותר - 186 קבוצות גברים ו-160 קבוצות נשים התייצבו לקו הזינוק. בסיום, הוכתרה אוזבקיסטן לאלופת הגברים ואוקראינה לאלופת הנשים.

מתכונת התחרות כללה 11 סיבובים, כשכל מפגש נערך על פני ארבעה לוחות. ההתמודדויות נקבעו על בסיס "השיטה השוויצרית", המשקללת באופן ממוחשב את חוזקה היחסי של המדינה ושל יריביה ואת הישגיה המצטברים. ניצחון קבוצתי - מינימלי (2.5-1.5) או מקסימלי (4:0) - מזכה את המדינה בשתי נקודות, תיקו קבוצתי בנקודה והפסד באפס. במקרה של שוויון נקודות קבוצתיות, מה שמכריע הוא עוצמת היריבים שמולם מתמודדת כל מדינה. כך קרה גם הפעם: אוזבקיסטן וארמניה הגיעו לסיבוב האחרון בשוויון נקודות עם 17 כל אחת, ושתיהן ניצחו בו 2.5-1.5 - ארמניה את ספרד ואוזבקיסטאן את הולנד. בזכות איכות הנקודות העדיפה הוענקה מדליית הזהב לאוזבקים, והכסף לארמנים.

ההישג של שתי הנבחרות מפתיע למדי. על הנייר דורגו האוזבקים במקום ה-14, אך שחקנם המוביל - נודירבק עבדוסתארוב בן ה-18, אלוף העולם בקצב משחק מהיר מאז 2021 - הצטיין וסחף גם את חבריו ליכולת מרשימה ביותר; הם היו הקבוצה היחידה שלא נוצחה באף מפגש. לארמנים, שבדירוג המוקדם הוצבו במקום ה-12, דווקא יש מסורת של הצטיינות אולימפית: בין 2008 ל-2012 הם זכו בטורניר שלוש פעמים ברצף. הפעם לא נחשבו לפייבוריטים, משום שהשחקן שהוליך אותם להישגים אלה - יהודי-למחצה בשם לבון ארוניאן - עבר לייצג ארצות הברית. למרות זאת, הלכידות הקבוצתית שהפגינו סייעה להם רבות: אף שהקרבות עצמם אינדיווידואליים, יש לה תפקיד חשוב בהפחתת התלהבות, בנכונות למאמץ מוגבר ובהשלמה עם החלטת המאמן על ארבעת המתחרים בכל סיבוב - משום שלכל קבוצה יש שחקן מילואים אחד.

  • הקבוצה השקיעה שנים בנהג מבטיח, רק כדי לקבל סטירות ממנו ומהיריבה הישירה
  • "אם תפעל מאוחר מדי, הנזק נעשה": בענף שמקדש סכנה, הם מנסים להקטין נזקים

הודו, בתור המארחת, העמידה שלוש קבוצות שונות. קבוצתה השנייה, שכללה ארבעה רבי אמנים בני 16 (!) ושחקן מילואים בן 30, זכתה במדליית הארד. הארבעה, כולם תושבי דרום הודו, הם התוצר הבולט ביותר של ההשקעה העצומה של המדינה בפיתוח הענף בשנים האחרונות - בהשראת אלוף העולם לשעבר וישי אנאנד, אף הוא תושב האזור. הפעם, בגיל 52, הוא כבר לא התחרה - אך העובדה ששלושה מהזוכים הגיעו ממדינת טאמיל נאדו, שצ'נאיי היא העיר הגדולה בה, תרמה להתלהבות האזורית והלאומית והובילה לקדחת שחמט של ממש.

העובדה ששלושה מהזוכים הגיעו מהמדינה שבה נמצאת צ'נאיי תרמה להתלהבות האזורית והלאומית, והובילה לקדחת שחמט של ממש

הסיפור הגדול והמרגש של האירוע היה הצעירים ההודים - וביניהם ניהאל סארין, ראמש באבו פראגנאנאנדה ורוונק סאדוואני - אך בראשם כולם עמד שחקן הלוח הראשון דימאראג'יו גוקש. עד הסיבוב לפני-האחרון הוא היה גיבור הקבוצה, עם שמונה ניצחונות בשמונה סיבובים - כולל מול כמה מגדולי תבל - ותיקו בסיבוב התשיעי. בקרב העשירי, הפך לגיבור טראגי: הוא השיג עמדה עדיפה מול עבדוסתארוב האוזבקי, כשתיקו היה מספיק לניצחון קבוצתי. אך גוקש חתר לניצחון אפשרי, שגה קשות והפסיד - מה שהביא לתיקו קבוצתי בלבד. הניצחון בסיבוב האחרון מול גרמניה הספיק רק למקום השלישי.

גם קבוצתה הראשונה של הודו הגיעה להישג יפה, ועם 17 נקודות סיימה במקום הרביעי. מתחתיה במקום החמישי, עם אותו מספר נקודות, התמקמה ארצות הברית. האמריקאים דורגו ראשונים על הנייר, אך סבלו מחולשתו של שחקנם החזק ביותר - סגן אלוף העולם לשעבר פביאנו קרואנה - וגם ארוניאן לא היה במיטבו. את הצמרת השלימו מולדובה, אף היא עם 17 נקודות, ולאחר מכן אזרבייג'אן, הונגריה, פולין וליטא עם 16 כל אחת. מגנוס קרלסן עמד בדיבורו להתחרות בלוח הראשון של נורווגיה וצבר 7.5 מ-9, אך קבוצתו סיימה רק במקום ה-59.

יש המאוכזבים מהישגי ישראל בתחרות, אך אלה בעיקר מי שהיו להם ציפיות מופרזות. בשורה התחתונה היא סיימה במקום ה-16 - גבוה יותר מהמקום ה-22 שהוענק לה בדירוג המוקדם על הנייר. בוריס גלפנד אמנם נעדר הפעם בשל סכסוך ארוך עם הנהלת האיגוד, אך גם איתו כשלה ישראל באולימפיאדה הקודמת. רק לאחד משחקניה, תמיר נבאתי, יש מד כושר של מעל 2600 - ואילו שאר חבריו פשוט אינם חזקים מספיק. הם לא התעלו בעימותים מול הקבוצות החזקות של ארצות הברית ואזרבייג'אן, וכשלו גם בהפסד מול גרמניה ובתיקו מול הפיליפינים. בסיכום התוצאות האישיות, השיג תמיר נבאתי 5 מ-9, אביטל בורוחובסקי 4 מ-9, איליה סמירין 5.5 מ-8 - תוצאה יפה בגיל 54 - מקסים רודשטיין 7 מ-10 ויבגני פוסטני 6.5 מ-8.

בתחרות הנשים בלטו שתי אחיות אצל המנצחת מאוקראינה, מריה ואנה מוזיצ'וק, והסבו לארצן קצת גאווה בימים קשים. האוקראיניות, שגברו בסיבוב האחרון על פולין, סיימו עם 18 נקודות - כמו גאורגיה - אך זכו בזהב בשל איכות הנקודות; הגאורגיות, שניצחו באותו סיבוב את אזרבייג'ן, הסתפקו בכסף. מי שנחלו אכזבה גדולה היו הקבוצה הראשונה של הודו, שהגיעה לסיבוב האחרון כשהיא מובילה ביתרון נקודה - אך נוצחה 3:1 מול האמריקניות והשיגה רק את הארד. גם ארצות הברית וקזחסטאן, עם 17 נקודות כל אחת, סיימו בצמרת.

בניגוד לגברים, אצל הנשים ישראל סיימה במקום נמוך יותר מבדירוג המוקדם - 19 לעומת 25, כשהפסד לבולגריה בסיבוב האחרון דרדר את מעמדה. בניקוד האישי השיגה מרסל אפרוימסקי 6.5 מ-10, דינה בלנקיה 5 מ-10, מאשה קלינובה 2.5 מ-5, מיכל להב 7 מ-10 ורונית לויטן 5.5 מ-9.
