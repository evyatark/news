בני האדם שוכללו במהלך מיליוני שנות אבולוציה להגיב למצבים מסוימים בלי לחשוב יותר מדי. אם אבות אבותיכם זיהו תנועה בעשב, הם רצו מיד. את השאלות הם שאלו אחר כך. בה בעת, היכולת לנתח ולתכנן היא חלק ממה שמבדיל בין בני האדם לחיות אחרות.

השאלה מתי לבטוח בתחושות הבטן שלנו ומתי להתלבט ולשאול את עצמנו שאלות — כלומר, אם לחשוב מהר או לאט, בשפתו של הפסיכולוג דניאל כהנמן — חשובה במשרד לא פחות מאשר בסוואנה.

גם לאינסטינקטים יש מקום. במצבים של שירות לקוחות, למשל, אינטואיציה לעתים מנחה אותנו טוב יותר איך להתנהג מאשר התסריט המוכן מראש

חשיבה המבוססת על שיקול דעת היא סימן ההיכר של מקום עבודה המנוהל היטב. מהלכים אסטרטגיים ודיוני תקציב בנויים על סבבי פגישות, מזכרים, נוסחות ומצגות. תהליכי עבודה נבנים מראש, בדיוק כדי למנוע את אותו שימוש בתגובות אינסטינקטיביות. מסינון עיוור של מועמדים לעבודה, שמשמיט פרטים אישיים שגורמים להטיה, ועד שימוש בטכניקות של קריאת תיגר (Red Teaming) על תוכניות של החברה — בדרכים כאלה, השיטתיות אמורה לגבור על הרפלקסים.

אבל גם לאינסטינקטים יש מקום. יש החלטות המחוברות יותר לתגובות רגשיות, ולכן קשה הרבה יותר לנתח אותן באופן שיטתי — למשל, אם קמפיין שיווק ישקף את המהות של החברה שלכם, או אם עובד מסוים עובד טוב עם אחרים בצוות. במצבים של שירות לקוחות, למשל, אינטואיציה לעתים מנחה אותנו טוב יותר איך להתנהג מאשר התסריט המוכן מראש.

הניסיון מחדד את האינסטיקטים

תחושות בטן הן משהו שאפשר לשפר, ומחקרים רבים מראים כי אינטואיציה משתפרת מאוד עם ניסיון. בניסוי ידוע אחד שנערך ב-2012, מתנדבים התבקשו לבחון מבחר של תיקי יד של מעצבים ולזהות אם הם מזויפים או לא. כמה מהם התבקשו להחליט לפי אינסטינקט, והאחרים התבקשו לשקול היטב את החלטתם ולבסס אותה. האינטואיציה עבדה טוב יותר עבור מי שהיו להם לפחות שלושה תיקי יד של מעצבים — הם ניחשו טוב יותר מהאחרים. ככל שאתם מומחים יותר — במקרה של המחקר, ככל שיש לכם יותר פריטים לבחון — כך האינסטינקטים שלכם טובים יותר.

כשכמויות מידע אינסופיות זורמות מכל עבר בארגון המודרני, הפיתוי לבקש עוד ניתוח ולהרהר עוד קצת הוא תופעה פסיכולוגית ידועה היטב בשם "הטלת צל מילולי", שמראה את הסכנה שבחשיבת יתר

עם זאת, הסיבה האמיתית לאמץ חשיבה מהירה היא — שהיא מהירה. קבלת החלטות אינסטינקטיבית לפעמים מהווה את הדרך היחידה להצליח לעבור את יום העבודה. חוקרים באוניברסיטת קורנל העריכו פעם כי אנשים מקבלים יותר מ-200 החלטות ביום, רק על מה יאכלו. מקום העבודה מורכב משרשרת של החלטות — חלקן גדולות ורבות מהן קטנות: איך לתעדף, מה לתעדף, מתי להתערב, ממי להימנע במעלית, וכעת גם מהיכן לעבוד מדי יום.

דוגמה אחת היא כשתיבת הדואר הנכנס מתמלאת באי־מיילים חדשים בתחילת היום. אין דרך לעבור ולקרוא את כולם בקפידה. האינטואיציה היא מה שעוזר לנו להחליט לאילו לענות, ואילו למחוק או לא לפתוח: כאלה שהם חלק ממשימות מתמשכות — נפתח מיד; הודעות מאנשים שאנחנו כפופים להם ישירות — נפתח מיד; תזכורות על אבטחת מידע ממחלקת התמיכה — ככל הנראה נמחק.

מסתכנים בהטלת צל

האינסטינקט פועל גם במקרים שבהם אנשים לא לגמרי ממוקדים במשימה — אולי הם עובדים על משהו אחר במהלך שיחת זום, או משחקים שחמט בטלפון, או סתם בוהים בתקרה. פתאום יש שקט, והם מבינים ששאלו אותם משהו, או שמצפים מהם להגיב. זוהי המקבילה התעסוקתית של עמידה מול אריה בערבה — מי שהברירה הטבעית בעדו יצליח לחלץ מפיו משפט כללי כמו, "הייתי רוצה להבין איך אנחנו מודדים הצלחה", ויקבל מלמולי הסכמה מכל אלה שחלמו כמוהו.

חשיבה מהירה לא נועדה רק להישרדות. היא יכולה לעזור לארגון כולו. כשכמויות מידע אינסופיות זורמות מכל עבר בארגון המודרני, הפיתוי לבקש עוד ניתוח ולהרהר עוד קצת הוא תופעה פסיכולוגית ידועה היטב בשם "הטלת צל מילולי", שמראה את הסכנה שבחשיבת יתר: אנשים עלולים יותר לא לזהות חשוד במסדר זיהוי, אם הם כתבו לפני כן תיאור מפורט מדי של החשוד. מנהלים לעתים סובלים מהטלת צל — הם חושבים יותר מדי על בעיה פשוטה, עד שהיא נהפכת לבעיה מורכבת.

אז מתי להשתמש באינטואיציה במקום העבודה? זה תלוי בשלל גורמים: האם מקבלי ההחלטות הם בעלי מומחיות אמיתית בתחום? האם זהו תחום שבו הרגש חשוב יותר מההיגיון? ומעל לכל, האם שווה לדחות את ההחלטה? חשיבה איטית נדרשת עבור החלטות גדולות, אבל חשיבה מהירה היא הדרך להפסיק למרוח זמן.
